1//! Generate error code documentation from the source of truth (error enums).
2//!
3//! This binary reads the error codes, descriptions, details, and help text
4//! directly from the `ParseError` and `SolverError` implementations via their
5//! `code()`, `description()`, `details()`, and `help()` methods.
6//!
7//! Run with:
8//! ```bash
9//! cargo run --bin generate_error_docs [OUTPUT_FILE]
10//! ```
11//! Defaults to `docs/ERROR_CODES.md`. Use `-` for stdout.
12
13use umiaq::errors::ParseError;
14use umiaq::solver::SolverError;
15use std::fs::File;
16use std::io::{self, Write, BufWriter};
17use std::env;
18
19/// Macro to generate error documentation for any error type
20/// with `code()`, `description()`, `details()`, `help()`, and `display_detailed()` methods
21macro_rules! generate_error_docs {
22    ($errors:expr, $writer:expr) => {
23        for error in $errors {
24            let code = error.code();
25            let description = error.description();
26            let details = error.details();
27            let help = error.help();
28
29            writeln!($writer, "### {}: {}\n", code, description)?;
30            writeln!($writer, "**Details:** {}\n", details)?;
31
32            if let Some(help_text) = help {
33                writeln!($writer, "**How to fix:**")?;
34                writeln!($writer, "```")?;
35                writeln!($writer, "{}", help_text)?;
36                writeln!($writer, "```\n")?;
37            }
38
39            writeln!($writer, "**Example error message:**")?;
40            writeln!($writer, "```")?;
41            writeln!($writer, "{}", error)?;
42            writeln!($writer, "```\n")?;
43
44            writeln!($writer, "**Detailed format:**")?;
45            writeln!($writer, "```")?;
46            writeln!($writer, "{}", error.display_detailed())?;
47            writeln!($writer, "```\n")?;
48
49            writeln!($writer, "---\n")?;
50        }
51    };
52}

54/// Helper to create all `ParseError` variants for documentation
55fn all_parse_error_variants() -> Vec<ParseError> {
56    vec![
57        ParseError::ParseFailure { s: "BAD(INPUT".to_string() },
58        // RegexError--create by attempting to compile an invalid regex
59        ParseError::RegexError(fancy_regex::Regex::new("(?P<invalid").unwrap_err()),
60        ParseError::EmptyForm,
61        ParseError::InvalidLengthRange { input: "10-5".to_string() },
62        ParseError::InvalidComplexConstraint { str: "bad constraint".to_string() },
63        ParseError::InvalidInput { str: "bad input".to_string(), reason: "bad input".to_string() },
64        // ParseIntError--create by parsing invalid integer
65        ParseError::ParseIntError("not_a_number".parse::<usize>().unwrap_err()),
66        ParseError::ContradictoryBounds { min: 5, max: 3 },
67        ParseError::InvalidCharsetRange('z', 'a'),
68        ParseError::DanglingCharsetDash,
69        ParseError::ConflictingConstraint {
70            var_char: 'A',
71            older: "|A|=3".to_string(),
72            newer: "|A|=5".to_string()
73        },
74        ParseError::ClauseParseError {
75            clause: "BAD(INPUT".to_string(),
76            source: Box::new(ParseError::InvalidInput { str: "BAD(INPUT".to_string(), reason: "illegal character '(' in pattern".to_string() })
77        },
78        ParseError::InvalidVariableName { var: "1".to_string() },
79        ParseError::InvalidLowercaseChar { invalid_char: 'X' },
80        ParseError::InvalidAnagramChars {

83        },
84        // NomError--use a common error kind
85        ParseError::NomError(nom::error::ErrorKind::Tag),
86    ]
87}

100fn main() -> io::Result<()> {
101    let args: Vec<String> = env::args().collect();
102
103    let output_path = if args.len() > 1 {
104        &args[1]
105    } else {
106        "docs/ERROR_CODES.md"
107    };
108
109    let mut writer: Box<dyn Write> = if output_path == "-" {
110        Box::new(BufWriter::new(io::stdout()))
111    } else {
112        let file = File::create(output_path)?;
113        Box::new(BufWriter::new(file))
114    };
115
116    writeln!(writer, "# Error Code Reference\n")?;
117    writeln!(writer, "**⚠️ This document is auto-generated from the source code. Do not edit manually.**\n")?;
118
119    writeln!(writer, "## Table of Contents\n")?;
120    writeln!(writer, "- [Solver Errors (S001–S003)](#solver-errors)")?;
121    writeln!(writer, "- [Parse Errors (E001–E016)](#parse-errors)")?;
122    writeln!(writer, "- [How to Use Error Codes](#how-to-use-error-codes)\n")?;
123
124    writeln!(writer, "## Solver Errors\n")?;
125    writeln!(writer, "Top-level errors from the solver. These wrap lower-level parse errors or indicate solver-specific issues.\n")?;
126    generate_error_docs!(all_solver_error_variants(), writer);
127
128    writeln!(writer, "## Parse Errors\n")?;
129    writeln!(writer, "Errors that occur when parsing pattern strings, constraints, or equations.\n")?;
130    generate_error_docs!(all_parse_error_variants(), writer);
131
132    writeln!(writer, "\n## How to Use Error Codes\n")?;
133    writeln!(writer, "When you see an error like:\n")?;
134    writeln!(writer, "```")?;
135    writeln!(writer, "Error: Empty form string (E003)")?;
136    writeln!(writer, "Example: Use 'A*B' or '*cat*' instead of empty string")?;
137    writeln!(writer, "```\n")?;
138    writeln!(writer, "1. Note the error code (e.g., `E003`)")?;
139    writeln!(writer, "2. Look it up in this document for detailed explanation")?;
140    writeln!(writer, "3. Follow the suggested resolution steps\n")?;
141
142    writeln!(writer, "## Error Display Formats\n")?;
143    writeln!(writer, "Errors are displayed in two formats:\n")?;
144    writeln!(writer, "### Simple Format")?;
145    writeln!(writer, "```")?;
146    writeln!(writer, "Error: <message>")?;
147    writeln!(writer, "```\n")?;
148    writeln!(writer, "### Detailed Format (via `display_detailed()`)")?;
149    writeln!(writer, "```")?;
150    writeln!(writer, "<message> (<code>)")?;
151    writeln!(writer, "<help text if available>")?;
152    writeln!(writer, "```\n")?;
153
154    Ok(())
155}

162    fn generate_docs_to_string() -> String {
163        use std::process::{Command, Stdio};
164
165        let output = Command::new("cargo")
166            .args(&["run", "--bin", "generate_error_docs", "--", "-"])
167            .stdout(Stdio::piped())
168            .stderr(Stdio::null())
169            .output()
170            .expect("Failed to run generate_error_docs");
171
172        String::from_utf8(output.stdout).expect("Invalid UTF-8 in generated docs")
173    }
