1use std::collections::HashSet;
2use std::ops::RangeInclusive;
3use std::sync::LazyLock;
4
5// Character-set constants
6pub(crate) const ALPHABET_SIZE: usize = 26;
7pub(crate) const LOWERCASE_ALPHABET: RangeInclusive<char> = 'a'..='z';
8#[cfg(test)]
9pub(crate) const UPPERCASE_ALPHABET: RangeInclusive<char> = 'A'..='Z';
10
11pub(crate) const VOWELS: &str = "aeiouy";
12pub(crate) const CONSONANTS: &str = "bcdfghjklmnpqrstvwxz";
13pub(crate) const VARIABLE_CHARS: &str = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
14pub(crate) const LITERAL_CHARS: &str = "abcdefghijklmnopqrstuvwxyz";
15
16pub(crate) const NUM_POSSIBLE_VARIABLES: usize = VARIABLE_CHARS.len();
17
18static VOWEL_SET: LazyLock<HashSet<char>> = LazyLock::new(|| VOWELS.chars().collect());
19static CONSONANT_SET: LazyLock<HashSet<char>> = LazyLock::new(|| CONSONANTS.chars().collect());
20
21pub(crate) trait UmiaqChar {
22    fn is_vowel(&self) -> bool;
23    fn is_consonant(&self) -> bool;
24    fn is_variable(&self) -> bool;
25    fn is_literal(&self) -> bool;
26}
27
28impl UmiaqChar for char {
29    fn is_vowel(&self) -> bool {
30        VOWEL_SET.contains(self)
31    }
32    fn is_consonant(&self) -> bool {
33        CONSONANT_SET.contains(self)
34    }
35    fn is_variable(&self) -> bool {
36        self.is_ascii_uppercase()
37    }
38    fn is_literal(&self) -> bool {
39        self.is_ascii_lowercase()
40    }
41}

86            let is_c = c.is_consonant();
87            assert_ne!(is_v, is_c, "char '{}' should be either vowel or consonant, not both or neither", c);
