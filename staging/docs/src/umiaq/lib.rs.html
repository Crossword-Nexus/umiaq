1//! Umiaq is a pattern-matching solver for crossword puzzles and similar word games.
2//!
3//! # Core Functionality
4//!
5//! The main entry point is [`solver::solve_equation`], which takes a pattern string and
6//! an entry list, returning matching solutions.
7//!
8//! # Error Handling
9//!
10//! Umiaq uses structured error types with error codes and helpful messages:
11//!
12//! - [`solver::SolverError`] (Top-level errors (S001-S003))
13//! - [`errors::ParseError`] (Parsing errors (E001-E016))
14//!
15//! ## Example: Basic Usage
16//!
17//! ```
18//! use umiaq::solver;
19//!
20//! let entries = vec!["cat", "dog", "catalog"];
21//! let result = solver::solve_equation("A*B", &entries, 10)?;
22//!
23//! for solution in result.solutions {
24//!     println!("{}", solver::solution_to_string(&solution)?);
25//! }
26//! # Ok::<(), Box<dyn std::error::Error>>(())
27//! ```
28//!
29//! ## Example: Error Handling with Detailed Messages
30//!
31//! ```
32//! use umiaq::solver::{self, SolverError};
33//!
34//! let entries = vec!["test"];
35//! let result = solver::solve_equation("", &entries, 10);
36//!
37//! match result {
38//!     Ok(r) => println!("Found {} solutions", r.solutions.len()),
39//!     Err(e) => {
40//!         // Display with error code and help text
41//!         eprintln!("Error: {}", e.display_detailed());
42//!         // Error code for documentation lookup
43//!         eprintln!("Code: {}", e.code());
44//!     }
45//! }
46//! ```
47//!
48//! ## Example: Handling Specific Error Types
49//!
50//! ```
51//! use umiaq::solver::{self, SolverError};
52//! use umiaq::errors::ParseError;
53//!
54//! let entries = vec!["test"];
55//! match solver::solve_equation("BAD(PATTERN", &entries, 10) {
56//!     Err(SolverError::ParseFailure(parse_err)) => {
57//!         // Access the specific ParseError
58//!         match &*parse_err {
59//!             ParseError::InvalidInput { str, reason } => {
60//!                 println!("Invalid input in '{}': {}", str, reason);
61//!             }
62//!             ParseError::ClauseParseError { clause, source } => {
63//!                 println!("Error in clause '{}': {}", clause, source);
64//!             }
65//!             _ => println!("Other parse error: {}", parse_err),
66//!         }
67//!     }
68//!     Err(e) => println!("Solver error: {}", e),
69//!     Ok(_) => println!("Success"),
70//! }
71//! ```
