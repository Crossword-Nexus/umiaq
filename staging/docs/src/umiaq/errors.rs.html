1//! Error types for parsing operations with error codes and helpful messages.
2//!
3//! # Error Codes
4//!
5//! Each error variant has a unique code (E001–E016) for documentation lookup:
6//!
7//! - E001: `ParseFailure` (Generic parse failure)
8//! - E002: `RegexError` (Invalid regex pattern)
9//! - E003: `EmptyForm` (Empty form string)
10//! - E004: `InvalidLengthRange` (Invalid length range format)
11//! - E005: `InvalidComplexConstraint` (Invalid complex constraint)
12//! - E006: `InvalidInput` (Invalid input)
13//! - E007: `ParseIntError` (Integer parsing error)
14//! - E008: `ContradictoryBounds` (Contradictory length bounds)
15//! - E009: `InvalidCharsetRange` (Invalid charset range)
16//! - E010: `DanglingCharsetDash` (Dangling '-' in charset)
17//! - E011: `ConflictingConstraint` (Conflicting variable constraints)
18//! - E012: `ClauseParseError` (Parse error in clause (wraps another error))
19//! - E013: `InvalidVariableName` (Variable name not A-Z)
20//! - E014: `InvalidLowercaseChar` (Non-lowercase character)
21//! - E015: `InvalidAnagramChars` (Invalid characters in anagram)
22//! - E016: `NomError` (Low-level nom parser error)
23//! - E017: `PrefilterFailed` (Regex prefilter failed during matching)
24//! - E018: `AnagramCheckFailed` (Anagram validation failed during matching)
25//! - E019: `UnsupportedConstraintType` (Constraint type not supported)
26//! - E020: `JointConstraintContradiction` (Joint constraint cannot be satisfied)
27//! - E021: `NegativeLengthConstraint` (Length constraint would be negative)
28//! - E022: `LengthConstraintExceedsMax` (Length constraint exceeds maximum representable value)
29//!
30//! # Examples
31//!
32//! ## Basic Error Handling
33//!
34//! ```
35//! use umiaq::errors::ParseError;
36//!
37//! fn parse_something(input: &str) -> Result<(), Box<ParseError>> {
38//!     if input.is_empty() {
39//!         return Err(Box::new(ParseError::EmptyForm));
40//!     }
41//!     Ok(())
42//! }
43//!
44//! match parse_something("") {
45//!     Err(e) => {
46//!         println!("Error: {}", e);
47//!         println!("Code: {}", e.code());
48//!         if let Some(help) = e.help() {
49//!             println!("Help: {}", help);
50//!         }
51//!     }
52//!     Ok(_) => println!("Success"),
53//! }
54//! ```
55//!
56//! ## Error Wrapping with Context
57//!
58//! ```
59//! use umiaq::errors::ParseError;
60//!
61//! fn parse_low_level(c: char) -> Result<usize, Box<ParseError>> {
62//!     if !c.is_ascii_lowercase() {
63//!         return Err(Box::new(ParseError::InvalidLowercaseChar { invalid_char: c }));
64//!     }
65//!     Ok(c as usize - 'a' as usize)
66//! }
67//!
68//! fn parse_high_level(entry: &str) -> Result<Vec<usize>, Box<ParseError>> {
69//!     entry.chars().map(|c| {
70//!         parse_low_level(c).map_err(|e| {
71//!             // Wrap with additional context
72//!             if let ParseError::InvalidLowercaseChar { invalid_char } = *e {
73//!                 Box::new(ParseError::InvalidAnagramChars {
74//!                     anagram: entry.to_string(),
75//!                     invalid_char
76//!                 })
77//!             } else {
78//!                 e
79//!             }
80//!         })
81//!     }).collect()
82//! }
83//! ```

90/// Custom error type for parsing operations
91#[derive(Debug, thiserror::Error)]
92pub enum ParseError {
93    #[error("Form parsing failed: \"{s}\"")]
94    ParseFailure { s: String },
95
96    #[error("Invalid regex pattern: {0}")]
97    RegexError(#[from] fancy_regex::Error),
98
99    #[error("Empty form string")]
100    EmptyForm,
101
102    #[error("Invalid length range: \"{input}\"")]
103    InvalidLengthRange { input: String },
104
105    #[error("{str}")]
106    InvalidComplexConstraint { str: String },
107
108    #[error("Invalid input in '{str}': {reason}")]
109    InvalidInput { str: String, reason: String },
110
111    #[error("int-parsing error: {0}")]
112    ParseIntError(#[from] ParseIntError),
113
114    #[error("contradictory bounds: min={min}, max={max}")]
115    ContradictoryBounds {
116        min: usize,
117        max: usize,
118    },
119
120    #[error("Invalid range in charset: {0}-{1}")]
121    InvalidCharsetRange(char, char),
122
123    #[error("Dangling '-' at end of charset")]
124    DanglingCharsetDash,
125
126    #[error("Conflicting constraints for {var_char} ({older} / {newer})")]
127    ConflictingConstraint { var_char: char, older: String, newer: String },
128
129    #[error("Parse error in clause '{clause}': {source}")]
130    ClauseParseError {
131        clause: String,
132        #[source]
133        source: Box<ParseError>,
134    },
135
136    #[error("Invalid variable name '{var}' (must be A-Z)")]
137    InvalidVariableName { var: String },
138
139    #[error("Invalid character '{invalid_char}' (only lowercase a-z allowed)")]
140    InvalidLowercaseChar { invalid_char: char },
141
142    #[error("Anagram constraint \"{anagram}\" contains invalid character '{invalid_char}' (only a-z allowed)")]
143    InvalidAnagramChars { anagram: String, invalid_char: char },
144
145    // // TODO use(?) (e.g., when detecting patterns with excessive nesting, backtracking potential)
146    // #[error("Pattern too complex: {reason}")]
147    // PatternComplexity { reason: String },
148
149    // nom parser error (lowest level)
150    #[error("nom parser error: {0:?}")]
151    NomError(ErrorKind),
152
153    // Matching errors (E017+)
154    #[error("Prefilter regex match failed: {0}")]
155    PrefilterFailed(#[source] fancy_regex::Error),
156
157    #[error("Anagram validation failed: {0}")]
158    AnagramCheckFailed(#[source] Box<ParseError>),
159
160    #[error("{constraint_type} is not supported")]
161    UnsupportedConstraintType { constraint_type: String },
162
163    #[error("Joint constraint {constraint} cannot be satisfied: {reason}")]
164    JointConstraintContradiction {
165        constraint: String,
166        reason: String,
167    },
168
169    #[error("Length constraint {constraint} is impossible (would be negative)")]
170    NegativeLengthConstraint { constraint: String },
171
172    #[error("Length constraint {constraint} exceeds maximum representable value")]
173    LengthConstraintExceedsMax { constraint: String },
174}

176impl From<ParseError> for io::Error {
177    fn from(pe: ParseError) -> Self {
178        // String version is the least fragile (no Send/Sync bounds issues)
179        io::Error::new(io::ErrorKind::InvalidInput, pe.to_string())
180    }
181}

195impl<'a> NomParseError<&'a str> for Box<ParseError> {
196    fn from_error_kind(_input: &'a str, kind: ErrorKind) -> Self {
197        Box::new(ParseError::NomError(kind))
198    }
199
200    fn append(_input: &'a str, _kind: ErrorKind, other: Self) -> Self {
201        // usually just return the existing error unchanged
202        other
203    }
204}

318    /// Formats the error with code and optional help text
319    #[must_use]
320    pub fn display_detailed(&self) -> String {
321        match self {
322            ParseError::ClauseParseError { clause: _, source } => {
323                // Flatten the report if it's a clause wrap to avoid redundancy
324                source.display_detailed()
325            }
326            _ => format_error_with_code_and_help(&self.to_string(), self.code(), self.help()),
327        }
328    }

331/// Helper function to format error messages with code and optional help text
332pub(crate) fn format_error_with_code_and_help(base_msg: &str, code: &str, help: Option<&str>) -> String {
333    if let Some(help_text) = help {
334        format!("{base_msg} ({code})\n{help_text}")
335    } else {
336        format!("{base_msg} ({code})")
337    }
338}
