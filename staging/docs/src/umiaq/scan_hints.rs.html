1// src/scan_hints.rs
2// -----------------------------------------------------------------------------
3// Fast, form-local length hints for prefiltering during the scan phase.
4//
5// This module computes length bounds for a single ParsedForm, taking into account:
6//   • fixed tokens in the form (i.e., literals, '.', '@', '#', [charset], /anagram)
7//   • frequencies of variables that appear in the form (Var, RevVar)
8//   • unary per-variable bounds from VarConstraints (normalized: min≥1, max is unbounded if unset)
9//   • joint group constraints from JointConstraints that refer ONLY to vars
10//     present in THIS form (e.g., "|AB|=6").
11//
12// The result can be used as a cheap prefilter: if a candidate entry's length does
13// not satisfy these bounds, you can skip calling the heavy matcher altogether.
14//
15// Design notes:
16// - We intentionally do *not* attempt global propagation across multiple forms.
17//   These hints are computed per-form, once per equation.
18// - We do not try to detect infeasibility of the full constraint set; if min>max
19//   emerges it simply means "no candidates" for that form.
20// - "!=" (not-equal) group relations are ignored for tightening because they do
21//   not produce a contiguous interval--we conservatively skip tightening on them.
22// - Presence of '*' in the form makes the form's max length unbounded for the
23//   hint's purposes (even if unary-var maxima are finite), because '*' can soak
24//   an arbitrary number of extra characters.
25// -----------------------------------------------------------------------------

33/// A joint constraint over variables *restricted to this form* considered
34/// as a contiguous total length bound for their raw lengths (not weighted).
35#[derive(Debug, Clone, PartialEq, Eq)]
36pub struct GroupLenConstraint {
37    pub vars: Vec<char>,          // e.g., ['A','B'] for |AB|
38    pub total_min: usize,         // inclusive
39    pub total_max: Option<usize>, // inclusive, None => unbounded
40}
41
42/// Resulting per-form hints.
43#[derive(Debug, Clone, PartialEq, Eq, Default)]
44pub struct PatternLenHints {
45    /// Lower bound on the form's length.
46    pub min_len: usize,
47    /// Upper bound on the form's length. `None` = unbounded above.
48    pub max_len_opt: Option<usize>,
49}
50
51/// Weighted row used in group constraint calculations.
52///
53/// - `w`: weight = number of times this variable appears in the form
54/// - `b`: per-variable bounds (`Bounds`)
55#[derive(Clone, Copy)]
56struct Row {
57    w: usize,
58    b: Bounds,
59}
60
61/// Per-form environment shared when applying group constraints.
62///
63/// This bundles everything that would otherwise need to be passed as
64/// multiple arguments into `tighten_with_group`, keeping the API clean
65struct FormContext<'a> {
66    /// All variables present in this form (sorted, deduped).
67    vars: &'a [char],
68    /// Frequency of each variable in this form (number of times it appears).
69    var_frequency: &'a HashMap<char, usize>,
70    /// Map of per-variable unary bounds for vars in this form.
71    bounds_map: &'a HashMap<char, Bounds>,
72    /// Total contribution of all fixed (non-variable) tokens in this form.
73    fixed_base: usize,
74    /// Global `VarConstraints` for the entire equation.
75    vcs: &'a VarConstraints,
76    /// Whether this form contains a `*` token (implies unbounded upper length).
77    has_star: bool,
78}
79
80impl FormContext<'_> {
81    /// Tighten the current `(weighted_min, weighted_max)` bounds with
82    /// respect to a single `GroupLenConstraint`.
83    ///
84    /// Returns the updated `(min, max)` pair.
85    fn tighten_with_group(
86        &self,
87        g: &GroupLenConstraint,
88        weighted_min: usize,
89        weighted_max: Option<usize>,
90    ) -> Bounds {
91        // Skip if group has no vars at all
92        if g.vars.is_empty() {
93            return weighted_max.map_or(Bounds::of_unbounded(weighted_min), |max| Bounds::of(weighted_min, max))
94        }
95
96        // Intersect with the form's variables (only consider vars that appear in this form)
97        let mut gvars: Vec<_> = g
98            .vars
99            .iter()
100            .copied()
101            .filter(|var_char| self.var_frequency.contains_key(var_char))
102            .collect();
103        gvars.sort_unstable();
104        if gvars.is_empty() {
105            return weighted_max.map_or(Bounds::of_unbounded(weighted_min), |max| Bounds::of(weighted_min, max))
106        }
107
108        // Build rows, Σ min_len, and Σ max_len (finite only)
109        let mut rows: Vec<Row> = Vec::with_capacity(gvars.len());
110        let mut sum_min_len = 0;
111        let mut sum_max_len_opt: Option<usize> = Some(0);
112        for &var_char in &gvars {
113            let b = self.bounds_map[&var_char];
114            rows.push(Row {
115                w: *self.var_frequency.get(&var_char).unwrap_or(&0),
116                b,
117            });
118            sum_min_len += b.min_len;
119            sum_max_len_opt = if b.max_len_opt.is_none() {
120                None
121            } else {
122                sum_max_len_opt.and_then(|a| b.max_len_opt.map(|u| a + u))
123            };
124        }
125
126        let weighted_min_for_t =
127            |t: usize| weighted_extreme_for_t(&rows, sum_min_len, sum_max_len_opt, t, Extreme::Min);
128        let weighted_max_for_t =
129            |t: usize| weighted_extreme_for_t(&rows, sum_min_len, sum_max_len_opt, t, Extreme::Max);
130
131        // ---- Account for group vars that are NOT in this form ------------------
132        // They eat into the group's total before we allocate to in-form vars.
133        // outside_form_min = Σ (min of vars outside the form)
134        // outside_form_max_opt = Σ (finite max of vars outside the form), None if any is unbounded
135        let (outside_form_min, outside_form_max_opt) = g
136            .vars
137            .iter()
138            .filter(|var_char| !self.var_frequency.contains_key(var_char))
139            .fold((0, Some(0)), |(min_acc, max_acc_opt), &var_char| {
140                let bounds = self.vcs.bounds(var_char);
141                let min_acc = min_acc + bounds.min_len;
142                let max_acc_opt = bounds.max_len_opt.and_then(|u| max_acc_opt.map(|a| a + u));
143                (min_acc, max_acc_opt)
144            });
145
146        // Effective totals for the in-form part of this group:
147        // - For the LOWER bound, outside takes as much as possible (use outside_form_max if finite).
148        // - For the UPPER bound, outside takes as little as possible (use outside_form_min).
149        // re 0: if outside can be arbitrarily large, in-form lower could be 0
150        let tmin_eff =
151            outside_form_max_opt.map_or(0, |of_max| g.total_min.saturating_sub(of_max));
152        let tmax_eff_opt = g
153            .total_max
154            .map(|tmax| tmax.saturating_sub(outside_form_min));
155
156        // Evaluate endpoints of the adjusted interval for in-form vars.
157        let gmin_w = weighted_min_for_t(tmin_eff);
158        let gmax_w = tmax_eff_opt.and_then(weighted_max_for_t);
159
160        // Combine with outside-of-group contributions (vars in this form but not in this group)
161        let outside: Vec<_> = self
162            .vars
163            .iter()
164            .copied()
165            .filter(|var_char| !gvars.contains(var_char))
166            .collect();
167
168        let outside_min = outside
169            .iter()
170            .map(|&var_char| *self.var_frequency.get(&var_char).unwrap_or(&0) * self.bounds_map[&var_char].min_len)
171            .sum::<usize>();
172
173        let outside_max_opt = if self.has_star {
174            None
175        } else {
176            outside.iter().copied().try_fold(0, |acc, var_char| {
177                self.bounds_map[&var_char].max_len_opt.map(|u| {
178                    acc + *self.var_frequency.get(&var_char).unwrap_or(&0) * u
179                })
180            })
181        };
182
183        let mut new_min = weighted_min;
184        let mut new_max = weighted_max;
185
186        if let Some(gmin) = gmin_w {
187            new_min = new_min.max(self.fixed_base + gmin + outside_min);
188        }
189
190        // Candidate upper bound from this group + outside
191        let candidate_upper = match (gmax_w, outside_max_opt) {
192            (Some(gm), Some(om)) => Some(self.fixed_base + gm + om),
193            _ => None,
194        };
195
196        if let Some(cand) = candidate_upper {
197            new_max = Some(new_max.map_or(cand, |cur| cur.min(cand)));
198        }
199
200        new_max.map_or(Bounds::of_unbounded(new_min), |max| Bounds::of(new_min, max))
201    }
202}

204/// Small enum for `weighted_extreme_for_t`
205#[derive(Clone, Copy, Debug)]
206enum Extreme { Min, Max }
207
208impl PatternLenHints {
209    /// Quick check for a candidate entry length against this hint.
210    pub(crate) fn is_entry_len_possible(&self, len: usize) -> bool {
211        len >= self.min_len && self.max_len_opt.is_none_or(|max_len| len <= max_len)
212    }
213}

215/// Convert a crate-level `JointConstraint` to a `GroupLenConstraint` interval.
216/// Returns `None` for incompatible/non-interval relations (e.g., NE) or empty.
217fn group_from_joint(jc: &JointConstraint) -> Option<GroupLenConstraint> {
218    // Map RelMask to [min,max] on the target.
219    // Note: For LT/GT we avoid underflow/overflow.
220    let (tmin, tmax_opt) = match jc.rel {
221        RelMask::EQ => (jc.target, Some(jc.target)),
222        RelMask::LE => (0, Some(jc.target)),
223        RelMask::LT => (0, Some(max(0, jc.target - 1))),
224        RelMask::GE => (jc.target, None),
225        RelMask::GT => (jc.target.saturating_add(1), None),
226        _ => return None // NE (or unusual mask combos) don't give a single interval — skip tightening.
227    };
228
229    // Basic sanity: empty interval ⇒ None
230    if let Some(tmax) = tmax_opt && tmin > tmax {
231        None
232    } else {
233        Some(GroupLenConstraint {
234            vars: jc.vars.clone(),
235            total_min: tmin,
236            total_max: tmax_opt,
237        })
238    }
239}

241/// Compute the weighted extremal sum at fixed total `t` over the rows,
242/// where each row contributes `w * len_i`, and `len_i ∈ [min_len, max_len]`.
243/// If `minimize` is true, distribute remaining length to cheaper weights first;
244/// otherwise to most expensive first.
245///
246/// `sum_min_len` is Σ `min_len`; `sum_max_len_opt` is Σ `max_len` if all `max_len` are finite, otherwise it is `None` (unbounded).
247fn weighted_extreme_for_t(
248    rows: &[Row],
249    sum_min_len: usize,
250    sum_max_len_opt: Option<usize>, // Some(sum_max_len) if ALL max_len are finite; None otherwise (i.e., if ANY max_len is None (i.e., unbounded))
251    t: usize,
252    extreme: Extreme,
253) -> Option<usize> {
254    // Feasibility checks
255    if t < sum_min_len {
256        return None;
257    }
258    if let Some(su) = sum_max_len_opt && t > su {
259        return None;
260    }
261
262    // Base cost at lower bounds
263    let base_weighted = rows.iter().map(|r| r.w.saturating_mul(r.b.min_len)).sum::<usize>();
264    let mut rem = t - sum_min_len;
265    if rem == 0 {
266        return Some(base_weighted);
267    }
268
269    // Greedy: assign remaining letters to cheapest (Min) or priciest (Max) first.
270    // We still honor each row's individual capacity (max_len - min_len). A row is "unbounded"
271    // iff r.max_len_opt is None.
272    let mut order: Vec<_> = rows.iter().collect();
273    match extreme {
274        Extreme::Min => order.sort_unstable_by_key(|r| r.w),                    // cheapest first
275        Extreme::Max => order.sort_unstable_by_key(|r| std::cmp::Reverse(r.w)), // priciest first
276    }
277
278    let mut extra = 0usize;
279    for r in order {
280        // Per-row capacity above min_len
281        let cap = if let Some(u) = r.b.max_len_opt {
282            u.saturating_sub(r.b.min_len).min(rem)
283        } else {
284            rem
285        };
286
287        if cap > 0 {
288            // Invariant: `cap <= rem` at each step (ensured by min() in calculation above)
289            debug_assert!(cap <= rem, "cap ({cap}) must be <= rem ({rem})");
290            extra = extra.saturating_add(r.w.saturating_mul(cap));
291            rem -= cap;
292            // Invariant: `rem` never goes negative because `cap <= rem` at each step.
293            // A debug_assert at the end checks that rem reaches 0 if t was feasible.
294            if rem == 0 {
295                break;
296            }
297        }
298    }
299
300    // If we reach here with rem != 0, the target length `t` is infeasible given the constraints.
301    // This is a normal outcome (not an error) when the combined min/max bounds make it impossible
302    // to construct a solution of exactly length `t`. For example, if variables have |A|=3, |B|=5,
303    // it's impossible to achieve total length 7. Returning None allows the caller to skip this
304    // length and try others.
305    if rem != 0 {
306        return None;
307    }
308
309    debug_assert_eq!(rem, 0, "Greedy allocation should have consumed all remaining length");
310    Some(base_weighted.saturating_add(extra))
311}

313/// Compute per-form hints from a `ParsedForm` *and* the equation's constraints.
314/// These are just length bounds for a parsed form
315///
316/// - `vcs`: the full equation's `VarConstraints` (we'll only read vars present in form)
317/// - `jcs`: the equation's `JointConstraints` (we'll filter to constraints whose
318///   variable set is a subset of the form's variables)
319pub(crate) fn form_len_hints_pf(
320    parts: &ParsedForm,
321    vcs: &VarConstraints,
322    jcs: &JointConstraints,
323) -> PatternLenHints {
324    // 1. Scan tokens: accumulate fixed_base, detect '*', and count var frequencies
325    let mut fixed_base = 0;
326    let mut has_star = false;
327    let mut var_frequency = HashMap::new();
328
329    for part in parts {
330        match part {
331            FormPart::Star => has_star = true,
332            FormPart::Dot | FormPart::Vowel | FormPart::Consonant | FormPart::Charset(_) => fixed_base += 1,
333            FormPart::Lit(s) => fixed_base += s.len(),
334            FormPart::Anagram(ag) => fixed_base += ag.len,
335            FormPart::Var(var_char) | FormPart::RevVar(var_char) => *var_frequency.entry(*var_char).or_insert(0) += VarConstraint::DEFAULT_MIN,
336        }
337    }
338
339    // Exact case (exit early): no variables and no star ⇒ exact fixed length
340    if var_frequency.is_empty() && !has_star {
341        return PatternLenHints {
342            min_len: fixed_base,
343            max_len_opt: Some(fixed_base),
344        };
345    }
346
347    // 2. Pull unary bounds just for vars in this form
348    let mut vars: Vec<_> = var_frequency.keys().copied().collect();
349    vars.sort_unstable();
350
351    let bounds_map = &vars
352        .iter()
353        .map(|&var_char| { (var_char, vcs.bounds(var_char)) })
354        .collect::<HashMap<char, Bounds>>();
355
356    let get_weight = |var_char: char| *var_frequency.get(&var_char).unwrap_or(&0);
357
358    // Baseline min/max ignoring groups
359    let mut weighted_min = {
360        let sum = vars
361            .iter()
362            .map(|&var_char| get_weight(var_char) * bounds_map[&var_char].min_len)
363            .sum::<usize>();
364        fixed_base + sum
365    };
366
367    // small helper function for maxes
368    let sum_weighted_max_len_opt = |vars: &[char]| -> Option<usize> {
369        if has_star {
370            return None;
371        }
372        vars.iter().copied().try_fold(0, |acc, var_char| {
373            bounds_map[&var_char].max_len_opt.map(|u| acc + get_weight(var_char) * u)
374        })
375    };
376
377    let sum_opt = sum_weighted_max_len_opt(&vars);
378    let mut weighted_max = sum_opt.map(|s| fixed_base + s);
379
380    // 3. Tighten with group constraints valid for this form
381    let ctx = FormContext {
382        vars: &vars,
383        var_frequency: &var_frequency,
384        bounds_map,
385        fixed_base,
386        vcs,
387        has_star,
388    };
389    for g in &group_constraints_for_form(parts, jcs) {
390        let new_bounds = ctx.tighten_with_group(g, weighted_min, weighted_max);
391        weighted_min = new_bounds.min_len;
392        weighted_max = new_bounds.max_len_opt;
393    }
394
395    PatternLenHints {
396        min_len: weighted_min,
397        max_len_opt: weighted_max,
398    }
399}
400
401/// Build the list of group constraints (as contiguous intervals) that are *scoped
402/// to this form*: every referenced variable must appear in the form.
403fn group_constraints_for_form(form: &ParsedForm, jcs: &JointConstraints) -> Vec<GroupLenConstraint> {
404    if jcs.is_empty() {
405        vec![]
406    } else {
407        let present: HashSet<_> = form.iter().filter_map(|p| match p {
408            FormPart::Var(var_char) | FormPart::RevVar(var_char) => Some(*var_char),
409            _ => None,
410        }).collect();
411
412        jcs.iter()
413            // ← revert to ANY overlap so constraints like |AB|=6 still inform A-only forms
414            .filter(|jc| jc.vars.iter().any(|var_char| present.contains(var_char)))
415            .filter_map(group_from_joint)
416            .collect()
417    }
418}

632        // sum_min_len = 3+2 = 5, so the group’s GE(4) is weaker than unary mins.

655    fn ranged_group_bounds_with_outside_var() {
656        // Form has only A,B. Group is |ABC| in [10, 12].
657        // C is outside the form with C in [3,4], so:
658        // - tmin_eff = 10 - max(C) = 10 - 4 = 6
659        // - tmax_eff = 12 - min(C) = 12 - 3 = 9
660        // A in [2,5], B in [1,7].

662        let mut vcs = VarConstraints::default();
663
664        let a = vcs.ensure('A');

668        b.bounds = Bounds::of(1, 7);
669
670        let c = vcs.ensure('C');
671        c.bounds = Bounds::of(3, 4);
672
