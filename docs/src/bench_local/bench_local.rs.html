1//! `bench_local.rs` — quick local timing runner (no Criterion)
2//!
3//! PURPOSE
4//! -------
5//! - Fast, ad-hoc timing for a handful of patterns on *your* machine.
6//! - Loads the entry list once, then runs each pattern several times and reports the median.
7//! - Always requests 100 results per pattern (by design, to keep comparisons simple).
8//! - Optionally shows a *static* time per pattern and the delta %.
9//!
10//! HOW TO RUN
11//! ----------
12//! - Optimized build:                `cargo run --bin bench_local --release`
13//! - Multiple repeats:               `cargo run --bin bench_local --release -- -r 5`
14//! - Print a few solutions:          `cargo run --bin bench_local --release -- -p 5`
15//! - See all flags:                  `cargo run --bin bench_local -- --help`
16//!
17//! NOTES
18//! -----
19//! - This is *not* Criterion. It's quick and convenient, not statistically rigorous.
20//! - Use the same machine and `--release` for more comparable numbers.
21//! - Patterns + optional static times live in `cases()` below.
22//! - I/O (printing) is kept outside the timed section.
23//! - One warm-up run per pattern is done (not included in timing).
24//! - We report the *median* over repeats (more robust than mean for small _N_).
25
26use clap::Parser;
27use std::hint::black_box;
28use std::time::Instant;
29use umiaq::bindings::Bindings;
30use umiaq::solver;
31use umiaq::entry_list;
32
33/// Simple local benchmark runner: load entry list once, time several patterns.
34/// Each case is a pattern + optional static time; name = pattern; always requests 100 results.
35#[derive(Parser, Debug)]
36#[command(author, version, about, long_about = None)]
37struct Cli {
38    /// Path to the entry list file (entry;score per line)
39    #[arg(
40        short,
41        long,
42        default_value = concat!(env!("CARGO_MANIFEST_DIR"), "/data/spreadthewordlist.dict")
43    )]
44    entry_list: String,
45
46    /// Minimum score filter
47    #[arg(short = 'm', long, default_value_t = 50)]
48    min_score: i32,
49
50    /// Number of repeats per pattern (use >1 to reduce noise; median is reported)
51    #[arg(short = 'r', long = "repeats", default_value_t = 1)]
52    num_repeats: usize,
53
54    /// Print up to this many solutions per pattern (0 = print none)
55    #[arg(short = 'p', long = "print", default_value_t = 0)]
56    print_limit: usize,
57}
58
59/// The fixed number of results we request per pattern.
60/// Keeping this constant across cases makes local comparisons simpler.
61const NUM_RESULTS: usize = 100;
62
63/// A benchmark case: the Umiaq pattern and an optional static time (seconds).
64/// Set `static_s` by running the query in an alternate tool against the Broda list.
65#[derive(Clone)]
66struct Case {
67    pattern: &'static str,
68    static_s: Option<f64>,
69}
70
71/// Edit/add new patterns here. The summary will display the pattern text as the "name".
72/// The `static_s` values below are results from an alternate tool
73fn get_cases() -> Vec<Case> {
74    vec![
75        Case { pattern: "AB;BA;|A|=2;|B|=2;!=AB", static_s: Some(0.260) },
76        // Note: this one is slow (for now?)
77        Case { pattern: "Atime;Btime;AB", static_s: Some(3.080) },
78        Case { pattern: "AB;CD;AC;BD", static_s: Some(0.300) },
79        Case { pattern: "A*BCD;AC*BD;DCB*A", static_s: Some(4.100) },
80        Case { pattern: "AkB;AlB", static_s: Some(2.360) },
81        Case { pattern: "l*x", static_s: Some(0.420) },
82        Case {
83            pattern: "ABCDEFGHIJKLMN;|ABCDEFGHIJKLMN|=14;!=ABCDEFGHIJKLMN",
84            static_s: Some(0.340)
85        },
86        Case { pattern: "A@B;A#B;!=AB;B=(g.*)", static_s: Some(24.420) },
87    ]
88}

90/// Small helper: robust central tendency for small samples.
91fn median(mut xs: Vec<f64>) -> f64 {
92    if xs.is_empty() {
93        return 0.0;
94    }
95    // safe: f64 durations are never NaN in this context
96    xs.sort_by(|a, b| a.partial_cmp(b)
97        .expect("f64 durations should not be NaN"));
98    let n = xs.len();
99    if n % 2 == 1 {
100        xs[n / 2]
101    } else {
102        0.5 * (xs[n / 2 - 1] + xs[n / 2])
103    }
104}

106const MAX_PATTERN_LEN: usize = 48;
107
108fn main() -> std::io::Result<()> {
109    /// One row in the benchmark summary: (pattern name, elapsed seconds,
110    /// number of results, optional reference time, optional other time).
111    type SummaryRow = (String, f64, usize, Option<f64>, Option<f64>);
112
113    let cli = Cli::parse();
114
115    // Load the entry list once. This I/O is *not* included in per-pattern timing.
116    eprintln!("Loading entry list from: {}", cli.entry_list);
117    let t_load = Instant::now();
118    let entry_list = entry_list::EntryList::load_from_path(&cli.entry_list, cli.min_score)?;
119    let load_secs = t_load.elapsed().as_secs_f64();
120    eprintln!("Loaded {} entries in {:.3}s", entry_list.entries.len(), load_secs);
121
122    // Keep references to avoid reallocating strings during benchmarks.
123    let entries_ref: Vec<_> = entry_list.entries.iter().map(String::as_str).collect();
124
125    let cases = get_cases();
126    // Store (pattern, median_seconds, solutions_last_run, static_s, delta_pct_opt) for the summary.
127    let mut summary: Vec<SummaryRow> = Vec::with_capacity(cases.len());
128
129    for (idx, case) in cases.iter().enumerate() {
130        let pattern = case.pattern;
131        eprintln!("\n[{:02}] {}", idx + 1, pattern);
132
133        // One *warm-up* execution per pattern to "touch" code paths / caches.
134        // We intentionally ignore its timing.
135        let _warmup = match solver::solve_equation(pattern, &entries_ref, NUM_RESULTS) {
136            Ok(result) => result,
137            Err(e) => {
138                eprintln!("  ✗ Warm-up failed: {e}");
139                continue;
140            }
141        };
142
143        // Repeat the timed runs and collect durations.
144        let mut times = Vec::with_capacity(cli.num_repeats);
145        let mut last_solutions: Vec<Vec<Bindings>> = Vec::new();
146
147        for rep in 0..cli.num_repeats {
148            // Keep only the *core* operation inside the timed region.
149            let t_solve = Instant::now();
150            let solve_result = match solver::solve_equation(black_box(pattern), &entries_ref, NUM_RESULTS) {
151                Ok(result) => result,
152                Err(e) => {
153                    eprintln!("  ✗ Run {}/{} failed: {}", rep + 1, cli.num_repeats, e);
154                    continue;
155                }
156            };
157            let solve_secs = t_solve.elapsed().as_secs_f64();
158
159            // Prevent the compiler from proving the result unused and eliding work.
160            let _keep = black_box(solve_result.solutions.len());
161
162            times.push(solve_secs);
163            last_solutions = solve_result.solutions;
164
165            eprintln!(
166                "  run {:>2}/{:>2}: {:.3}s ({} solutions)",
167                rep + 1,
168                cli.num_repeats,
169                solve_secs,
170                last_solutions.len()
171            );
172        }
173
174        // Prefer median for small N--it's less sensitive to noisy outliers.
175        let med = median(times);
176
177        // Optionally print a few solutions from the *last* run (outside timing).
178        if cli.print_limit > 0 && !last_solutions.is_empty() {
179            for sol in last_solutions.iter().take(cli.print_limit) {
180                let display = solver::solution_to_string(sol)?;
181                println!("{display}");
182            }
183        }
184
185        // Compute delta % vs. static value, if provided.
186        let delta_pct = case.static_s.and_then(|exp| {
187            if exp > 0.0 {
188                Some((med - exp) / exp * 100.0)
189            } else {
190                None
191            }
192        });
193
194        eprintln!(
195            "  → median {:.3}s over {} run(s); last run produced {} {}.{}",
196            med,
197            cli.num_repeats,
198            last_solutions.len(),
199            pluralizer(last_solutions.len(), "solution".into(), None),
200            match (case.static_s, delta_pct) {
201                (Some(exp), Some(dp)) => format!(" (static {exp:.3}s, Δ = {dp:+.1}%)"),
202                _ => String::new(),
203            }
204        );
205
206        summary.push((
207            pattern.to_string(),
208            med,
209            last_solutions.len(),
210            case.static_s,
211            delta_pct,
212        ));
213    }
214
215    // Compact summary at the end for a quick scan across all patterns.
216    eprintln!("\n==== Summary ====");
217    eprintln!(
218        "{:<MAX_PATTERN_LEN$} | {:>10} | {:>11} | {:>10} | {:>8}",
219        "pattern", "median (s)", "# solutions", "static (s)", "Δ %"
220    );
221    eprintln!(
222        "{:-<MAX_PATTERN_LEN$}-+-{:-<10}-+-{:-<11}-+-{:-<10}-+-{:-<8}",
223        "", "", "", "", ""
224    );
225    for (pat, med, num_solutions, static_t, delta_pct) in &summary {
226        // Trim very long patterns for readability in the summary.
227        let display = if pat.len() > MAX_PATTERN_LEN {
228            // "- 1" for the "…"
229            format!("{}…", pat.chars().take(MAX_PATTERN_LEN - 1).collect::<String>())
230        } else {
231            pat.clone()
232        };
233        let static_str = static_t.map(|x| format!("{x:.1}")).unwrap_or_else(|| "—".into());
234        let dp_str = delta_pct
235            .map(|x| format!("{x:+.1}"))
236            .unwrap_or_else(|| "—".into());
237        eprintln!(
238            "{display:<MAX_PATTERN_LEN$} | {med:>10.3} | {num_solutions:>11} | {static_str:>10} | {dp_str:>8}"
239        );
240    }
241
242    Ok(())
243}

245// TODO? put this elsewhere
246fn pluralizer(count: usize, singular: String, plural: Option<String>) -> String {
247    if count == 1 {
248        singular
249    } else {
250        plural.unwrap_or_else(|| singular + "s")
251    }
252}

267        assert_eq!(pluralizer(99, "radius".into(), Some("radii".into())), "radii");
